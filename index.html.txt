        // Initialize Lucide icons
        lucide.createIcons();

        // Mobile menu toggle
        const mobileMenuButton = document.getElementById('mobile-menu-button');
        const mobileMenu = document.getElementById('mobile-menu');
        mobileMenuButton.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });
        const mobileMenuLinks = mobileMenu.getElementsByTagName('a');
        for (let link of mobileMenuLinks) {
            link.addEventListener('click', () => {
                mobileMenu.classList.add('hidden');
            });
        }

        // --- Interactive Gallery Carousel ---
        const carouselContainer = document.querySelector('.carousel-container');
        if (carouselContainer) {
            const carouselPlacers = document.querySelectorAll('.carousel-placer');
            const itemCount = carouselPlacers.length;
            const angle = 360 / itemCount;
            const radius = 450; // Increased radius for a larger circle

            let isDragging = false;
            let startX;
            let currentRotation = 0;
            const rotationSpeed = 0.2;

            // Position placers in a circle and set up items
            carouselPlacers.forEach((placer, i) => {
                const itemAngle = angle * i;
                placer.style.transform = `rotateY(${itemAngle}deg) translateZ(${radius}px)`;
                
                const item = placer.querySelector('.carousel-item');
                item.dataset.angle = itemAngle;
                const counterRotation = -itemAngle;
                item.style.setProperty('--counter-rotation-deg', `${counterRotation}deg`);
            });

            const updateItemRotation = () => {
                 carouselPlacers.forEach(placer => {
                    const item = placer.querySelector('.carousel-item');
                    const itemAngle = parseFloat(item.dataset.angle);
                    const counterRotation = -currentRotation - itemAngle;
                    item.style.setProperty('--counter-rotation-deg', `${counterRotation}deg`);
                });
            };
            
            const startDrag = (e) => {
                isDragging = true;
                startX = e.pageX || e.touches[0].pageX;
                carouselContainer.style.transition = 'none';
            };

            const drag = (e) => {
                if (!isDragging) return;
                e.preventDefault();
                const x = e.pageX || e.touches[0].pageX;
                const deltaX = x - startX;
                currentRotation += deltaX * rotationSpeed;
                carouselContainer.style.transform = `rotateY(${currentRotation}deg)`;
                updateItemRotation();
                startX = x;
            };

            const stopDrag = () => {
                if (!isDragging) return;
                isDragging = false;
                carouselContainer.style.transition = 'transform 0.5s ease-out';
            };

            carouselContainer.addEventListener('mousedown', startDrag);
            document.addEventListener('mousemove', drag);
            document.addEventListener('mouseup', stopDrag);
            document.addEventListener('mouseleave', stopDrag);

            carouselContainer.addEventListener('touchstart', startDrag, { passive: true });
            document.addEventListener('touchmove', drag);
            document.addEventListener('touchend', stopDrag);
            
            // Initial rotation update
            updateItemRotation();
        }

        // Hero Section Fade-in Animation
        window.addEventListener('load', () => {
            setTimeout(() => {
                const heroContent = document.getElementById('hero-content');
                if (heroContent) {
                    heroContent.classList.add('animate-fade-in-left');
                    heroContent.classList.remove('fade-in-start');
                }
            }, 2000);
        });

    </script>
